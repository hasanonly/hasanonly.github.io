    cx.beginPath(); cx.arc(c.x,c.y,3,0,Math.PI*2); cx.fillStyle='#f0ece4'; cx.fill();
  });
}

function drawPart() {
  particles.forEach(p=>{ cx.globalAlpha=p.life; cx.beginPath(); cx.arc(p.x,p.y,p.r,0,Math.PI*2); cx.fillStyle=p.color; cx.fill(); });
  cx.globalAlpha=1;
}

function update() {
  frame++; score++;
  speed=4.5+Math.floor(score/300)*0.35;
  player.vy+=0.72; player.y+=player.vy;
  if(player.y>=GY){player.y=GY;player.vy=0;player.jumps=0;}

  nextObs--;
  if(nextObs<=0){
    const sz=22+Math.random()*18;
    obstacles.push({x:W+20,y:GY+player.h/2-sz*0.1,sym:SYMS[Math.floor(Math.random()*SYMS.length)],size:sz,color:PUR[Math.floor(Math.random()*PUR.length)],w:sz*0.6,h:sz});
    nextObs=Math.max(40,85+Math.random()*55-Math.floor(score/500)*5);
  }

  nextCol--;
  if(nextCol<=0){
    collectibles.push({x:W+10,y:GY-45-Math.random()*75,ph:Math.random()*Math.PI*2});
    nextCol=55+Math.random()*40;
  }

  obstacles.forEach(o=>o.x-=speed);
  obstacles=obstacles.filter(o=>o.x>-60);

  collectibles=collectibles.filter(c=>{
    c.x-=speed;
    const dx=(player.x+player.w/2)-c.x, dy=(player.y+player.h/2)-c.y;
    if(Math.sqrt(dx*dx+dy*dy)<22){ score+=50; burst(c.x,c.y,'#d4a847',8); soundCollect(); return false; }
    return c.x>-20;
  });

  particles.forEach(p=>{p.x+=p.vx;p.y+=p.vy;p.vy+=0.12;p.life-=0.04;});
  particles=particles.filter(p=>p.life>0);

  // Collision
  const px=player.x+6,py=player.y+5,pw=player.w-10,ph=player.h-8;
  for(let o of obstacles){
    const ox=o.x-o.w*0.38, oy=o.y-o.h*0.78;
    if(px<ox+o.w*0.76&&px+pw>ox&&py<oy+o.h&&py+ph>oy){ burst(player.x+13,player.y+13,'#7c6fff',22); soundDie(); gameState='dead'; return; }
  }
}

function updateHUD() {
  document.getElementById('hudScore').textContent=score;
  document.getElementById('hiScore').textContent=Math.max(hiScore,score);
}

function loop() {
  drawBg(); drawGround();
  if(gameState==='running') update();
  drawCollect(); drawObs(); drawPlayer(); drawPart(); updateHUD();
  if(gameState==='dead'){onDeath();return;}
  animId=requestAnimationFrame(loop);
}

function onDeath() {
  cancelAnimationFrame(animId);
  if(score>hiScore){
    hiScore=score; localStorage.setItem('aiq_hi',hiScore);
    document.getElementById('modalScore').textContent=score;
    document.getElementById('modal').classList.add('show');
  } else {
    document.getElementById('ovTitle').textContent='game over';
    document.getElementById('ovSub').textContent=`score: ${score} · keep going`;
    document.getElementById('playBtn').textContent='↺ try again';
    document.getElementById('gameOverlay').classList.remove('hidden');
    gameState='idle';
  }
}

function startGame() {
  resetGame(); gameState='running';
  document.getElementById('gameOverlay').classList.add('hidden');
  loop();
}

function seedIfEmpty(db) {
  const ref = db.ref('leaderboard');
  ref.once('value', snap => {
    if (!snap.exists()) {
      SEED.forEach(e => {
        const key = e.name.toLowerCase().replace(/[^a-z0-9_]/g,'_');
        ref.child(key).set({ name: e.name, score: e.score, subject: e.subject });
      });
    }
  });
}

function useFallback() {
  // Local-only mode with seed data
  SEED.forEach(e => { lbData[e.name.toLowerCase()] = e; });
  renderLB();
}

function submitToLB(name, score, subject) {
  const key = name.toLowerCase().replace(/[^a-z0-9_]/g,'_');
  const entry = { name, score, subject };

  if (db) {
    // Only update if this score is higher
    db.ref(`leaderboard/${key}`).transaction(current => {
      if (!current || score > current.score) return entry;
      return; // abort if not higher
    });
  } else {
    // Fallback local
    if (!lbData[key] || score > lbData[key].score) {
      lbData[key] = entry;
      renderLB();
    }
  }
}
